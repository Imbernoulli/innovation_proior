                if (res_val == target_dist - cand_dist) {
                    // Found the parent
                    edges.push_back(make_tuple(cand_id, u_id, (int)(target_dist - cand_dist)));
                    found = true;
                    break;
                }
            }
            // If not found, logic failure (should not happen with valid tree input)
        }

        // Output the result
        cout << "! ";
        for (size_t k = 0; k < edges.size(); ++k) {
            cout << get<0>(edges[k]) << " " << get<1>(edges[k]) << " " << get<2>(edges[k]);
            if (k != edges.size() - 1) cout << " ";
        }
        cout << "\n";
        cout.flush();
    }
    return 0;
}
